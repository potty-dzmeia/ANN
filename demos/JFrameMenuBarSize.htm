indeed...
	/**
	 * Sets the size of this frame such that the inner content pane is the 
	 * specified size.  This attempts to take the frame border insets and 
	 * menubar height into account.  <br />
	 * <br />
	 * For example:  A typical JFrame on Windows has a 23 pixel high 
	 * titlebar, a 23 pixel high menubar (if visible) and 4 pixel thick 
	 * border on the other 3 sides.  So setting an inner size of 500 x 400 
	 * would set the actual frame's size to 508 x 450 (or 508 x 427 if no 
	 * menubar is set).  
	 * 
	 * @param  dim  the size
	 * @see  #setInnerSize(int, int)
	 */
	public static void setInnerSize(JFrame frame, Dimension dim) {
		JMenuBar menubar = frame.getJMenuBar();
		int mbh = (menubar != null ? menubar.getSize().height : 0);
		Insets insets = frame.getInsets();
		dim.width += insets.left + insets.right;
		dim.height += insets.top + insets.bottom + mbh;
		frame.setSize(dim);
	}
